7.4 Grammar (dplyr vs SQL)

7.4.1 select()

dplyr SQL 
# all columns
students
# select name and grade
students %>% 
  select(name, gender)
 
-- select all columns
SELECT * FROM students
-- select name and grade
SELECT name, grade FROM students

7.4.2 mutate()

dplyr SQL 
# new variable: grade_percent
# expressing grade out of 100
students %>% 
  mutate(
    grade_percent = grade/20 * 100
  )
 
-- new variable: grade_percent
-- expressing grade out of 100
SELECT (grade/20.0 * 100)
AS grade_percent
FROM students;

7.4.3 filter()

dplyr SQL 
# keep if gender is female
# or student is Sheldon
students %>% 
  filter(
    name == 'Sheldon' |
    gender == 'Female'
  )

7.4.4 summarise() with group_by()

dplyr SQL 
# grouped by gender,
# find average grade
students %>% 
  group_by(gender) %>% 
  summarise(avg_grade = mean(grade))
 
-- grouped by gender,
-- find average grade
SELECT gender, AVG(grade)
AS avg_grade
FROM students
GROUP BY gender;

7.4.5 arrange()

dplyr SQL 
# sort name 
# in descending order
students %>% 
  arrange(desc(name))
 
-- sort name 
-- in descending order
SELECT *
FROM students
ORDER BY name DESC;
