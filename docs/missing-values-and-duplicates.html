6.1 Missing Values

6.1.1 tidyr::drop_na() vs pd.dropna() and pd.notpna()

name age
A 21
B NA
NA NA
D 40

How to drop missing rows or columns?

tidyr::drop_na() pd.dropna() and pd.notna() 
# toy data
df <- tibble(
  name = c('A', 'B',
           NA, 'D'),
  age  =  c(21, NA,
           NA, 40)
)
df
## # A tibble: 4 x 2
##   name    age
##   <chr> <dbl>
## 1 A        21
## 2 B        NA
## 3 <NA>     NA
## 4 D        40
# drop missing rows
df %>% drop_na()
## # A tibble: 2 x 2
##   name    age
##   <chr> <dbl>
## 1 A        21
## 2 D        40
# drop specific variable
# with NaN
df %>% drop_na(name)
## # A tibble: 3 x 2
##   name    age
##   <chr> <dbl>
## 1 A        21
## 2 B        NA
## 3 D        40
 
# toy data
df = pd.DataFrame({
    'name': [
        'A', 'B', np.nan, 'D'
    ],
    'age': [
        21, np.nan, np.nan, 40
    ]    
})
df
##   name   age
## 0    A  21.0
## 1    B   NaN
## 2  NaN   NaN
## 3    D  40.0
# drop missing rows
df.dropna()
##   name   age
## 0    A  21.0
## 3    D  40.0
# drop specific variable
# with NaN
df[df['name'].notna()]
##   name   age
## 0    A  21.0
## 1    B   NaN
## 3    D  40.0
